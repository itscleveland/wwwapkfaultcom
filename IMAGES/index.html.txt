  <script>
    const messages = [
  "Premium versions of your favorite apps.",
  "Faster speeds, fewer bugs, smoother operation.",
  "Exclusive features not found in the original apps.",
  "Ad-free experience.",
  "Stay current with the latest, most powerful app versions."
];


    const dynamicText = document.getElementById('dynamicText');
    const loadingScreen = document.getElementById('loadingScreen');
    const radialBurst = document.getElementById('radialBurst');

    let currentMessage = 0;

    // Function to update text
    function updateText() {
      if (currentMessage < messages.length) {
        dynamicText.textContent = messages[currentMessage];
        currentMessage++;
        setTimeout(updateText, 2000); // Change message every 2 seconds
      } else {
        endLoading();
      }
    }

    // Function to handle end of loading
    function endLoading() {
      // Trigger radial burst animation
      radialBurst.style.opacity = '1';
      radialBurst.style.animation = 'burst 1s ease-out forwards';

      // Fade out loading screen
      loadingScreen.style.opacity = '0';
      setTimeout(() => {
        loadingScreen.style.display = 'none';
        // Redirect to a new page or replace current content
        window.location.href = "https://itscleveland.github.io/wwwapkfaultcom/"; // Replace with your URL
      }, 1000);
    }

    // Start the text animation on window load
    window.onload = () => {
      updateText();
    };
